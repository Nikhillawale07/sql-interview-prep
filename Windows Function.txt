select s.*,
max(fees) over() as Max_Salary
from student s;

select s.*,
max(fees) over(partition by dept) as Max_Salary
from student s;

select s.*,
row_number() over() as rn
from student s;

select s.*,
row_number() over(partition by dept) as rn
from student s;

select * from(
select s.*,
row_number() over(partition by dept order by Id) as rn
from student s) x
where x.rn<3;

select Name,fees,
RANK() OVER(ORDER BY fees desc) AS feeRank
from student s;

select * from(
select s.*,
RANK() OVER(partition by dept order by fees desc) as rnk
from student s) x
where x.rnk<4;





/*DENSE_RANK*/

    DENSE_RANK() OVER (ORDER BY fees DESC) AS SalaryRank
